class tinyJquery {
  constructor(name) {
    if (typeof name == 'string') {
      this.dom = document.querySelectorAll(name)
    } else if (name.constructor.name == 'NodeList' || Array.isArray(name)) {
      this.dom = name
    } else {
      this.dom = [name]
    }
  }
}
